func makeRandomBooleanValue () -> Bool {
    let randomNum = Int(arc4random_uniform(50))
    guard randomNum % 2 == 0 else { return false }
    return true
}

            if input[outer][inner] == true {
                print("|", terminator: "-")
            } else {
                print("|", terminator: " ")
            }
        }

    }
